Bad way

function innerHtmlBadWay(){
	for (var i = 0; i < 10000; i++){
		document.getElementById('myDiv').innerHTML += 'x';
	}
}
innerHtmlBadWay();

Good way

function innerHtmlGoodWay(){
	var content = 'x';
	for (var i = 0; i < 10000; i++){
		content += 'x';
	}
	document.getElementById('myDiv').innerHTML += content;
}
innerHtmlGoodWay();
